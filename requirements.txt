# xsphere slow control — Python dependencies
# Install with:  pip install -r requirements.txt
#
# Then install the LabJack plugin in editable mode:
#                pip install -e LJ-python-controller/
#
# NOTE: The LabJack T7 hardware also requires the LJM runtime library.
#       Download and install it from https://labjack.com/pages/support
#       before connecting to a device.

paho-mqtt>=2.0.0        # MQTT broker client

pyyaml>=6.0             # YAML config loader (config.yaml, state.yaml)
numpy>=1.24.0           # Numerical utilities

# ── Web control panel (webcontrol/) ────────────────────────────────────────────
flask>=3.0              # the GUI server
waitress                # production-grade WSGI server (optional; Flask dev otherwise)
markdown>=3.0           # renders README.md at /readme

# ── LabJack T7 ────────────────────────────────────────────────────────────────
labjack-ljm             # LJM Python wrapper (requires LJM runtime installed separately)
